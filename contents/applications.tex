\chapter{Aplicaciones}
\section{Teorema de Eckart-Young}
Una de las aplicaciones más potentes de la descomposición en valores singulares (SVD) radica en su extraordinaria capacidad para comprimir información de manera eficiente, reduciendo el espacio de almacenamiento sin perder las características esenciales de los datos. Para abrir este bloque práctico, presentaremos un teorema fundamental que demuestra rigurosamente por qué las aproximaciones de bajo rango obtenidas al truncar la SVD son óptimas bajo ciertas normas matriciales; un principio teórico que, acto seguido, pondremos en práctica mediante un ejemplo concreto enfocado en la compresión de imágenes digitales.
\begin{definition}
	Se define la siguiente forma bilineal en el espacio de matrices:
	\[ \ip{A, B} = \tr(A^{\top} B). \]
\end{definition}

\begin{proposition}
	La anterior forma bilineal es un producto escalar \textit{(llamado producto de Frobenius)}.
\end{proposition}

\begin{proof}
	Es necesario ver las tres propiedades que cumple un producto escalar:
	\begin{enumerate}
		\item Linealidad con respecto al segundo argumento:
		      \begin{align*}
			      \ip{A, B+C} & = \tr(A^{\top} (B+C)) = \tr(A^{\top} B + A^{\top} C)       \\
			                  & = \tr(A^{\top} B) + \tr(A^{\top} C) = \ip{A,B} + \ip{A,C}.
		      \end{align*}
		\item Verificamos que es simétrica. \\
		      Como $\tr(X) = \tr(X^{\top})$, entonces $\tr(A^{\top}B) = \tr\parens*{(A^{\top}B)^{\top}} = \tr(B^{\top} A) = \ip{B, A}$.
		\item Vemos que $\ip{A, A} \geq 0$ y que $\ip{A, A} = 0 \iff A = 0$. \\
		      Sea $A = (a_{ij}) \in M_{m \times n}(\R)$. Por definición de la multiplicación matricial, el elemento en la posición $(i, i)$ de la matriz $A^{\top}A$ viene dado por el producto escalar de la fila $i$ de $A^{\top}$ (que es la columna $i$ de $A$) por la columna $i$ de $A$:
		      \[ (A^{\top} A)_{ii} = \sum_{k=1}^{m} (A^{\top})_{ik} a_{ki} = \sum_{k=1}^{m} a_{ki}^2. \]
		      Calculando la traza, que es la suma de los elementos de la diagonal principal, se obtiene:
		      \[ \ip{A, A} = \tr(A^{\top}A) = \sum_{i=1}^{n} (A^{\top} A)_{ii} = \sum_{i=1}^{n} \sum_{k=1}^{m} a_{ki}^2. \]
		      Dado que $a_{ki}^2 \geq 0$ para todo $k, i$, se tiene inmediatamente que $\ip{A, A} \geq 0$.

		      Además, una suma de términos no negativos es igual a cero si y solo si cada uno de sus términos es cero. Por lo tanto:
		      \[ \tr(A^{\top}A) = 0 \iff a_{ki}^2 = 0 \quad \forall k, i \iff a_{ki} = 0 \quad \forall k, i \iff A = 0, \]
		      que concluye la demostración.



	\end{enumerate}
\end{proof}

\begin{definition}
	Se define la norma de Frobenius inducida por el producto escalar como:
	\[ \norm{A}_F = \sqrt{\ip{A, A}} = \sqrt{\sum_{i=1}^{m}\sum_{j=1}^{n} |a_{ij}|^2}, \]
	con $A = (a_{ij}) \in M_{m \times n}(\R)$.
\end{definition}

\begin{definition}
	Se define la norma espectral de $A$ inducida por el producto escalar usual como:
	\[ \norm{A}_2 = \sup_{x \neq 0} \frac{\norm{Ax}_2}{\norm{x}_2} = \max_{\norm{x}_2 = 1} \norm{Ax}_2. \]
\end{definition}

La verificación es inmediata a partir de las propiedades de la norma vectorial $\norm{\cdot}_2$, por lo que se omite.

\begin{lemma}\label{lem:normas-svd}
	Sea $A \in M_{m \times n}(\R)$ de rango $r$ y sea $A = U \Sigma V^{\top}$ su descomposición en valores singulares. Entonces:
	\begin{enumerate}
		\item $\norm{A}_2 = \sigma_1$.
		\item $\norm{A}_F = \sqrt{\sigma_1^2 + \ldots + \sigma_r^2}$.
	\end{enumerate}
\end{lemma}

\begin{proof}
	\begin{enumerate}
		\item
		      Por definición de la norma inducida por la norma euclídea,
		      \[
			      \|A\|_2 \;=\; \max_{\|x\|_2=1}\|Ax\|_2.
		      \]

		      Sea la descomposición en valores singulares
		      \[
			      A = U \Sigma V^{\top},
		      \]
		      con \(U \in O(m)\), \(V \in O(n)\) y
		      \(\Sigma=\mathrm{diag}(\sigma_1,\sigma_2,\dots,\sigma_r,0,\dots)\),
		      donde \(\sigma_1\ge \sigma_2 \ge \cdots \ge \sigma_r>0\).

		      Para \(x\) tal que \(\|x\|_2=1\), definimos \(y=V^{\top} x\). Como \(V\) es ortogonal,
		      \(\|y\|_2=\|x\|_2=1\). Entonces
		      \[
			      \|Ax\|_2
			      = \|U\Sigma V^{\top} x\|_2
			      = \|U\Sigma y\|_2
			      = \|\Sigma y\|_2,
		      \]
		      ya que \(U\) preserva la norma.

		      Escribiendo \(y=(y_1,\dots,y_n)\), se tiene
		      \[
			      \|\Sigma y\|_2^2
			      = \sigma_1^2 y_1^2 + \sigma_2^2 y_2^2 + \cdots + \sigma_r^2 y_r^2
			      \le \sigma_1^2 (y_1^2+\cdots+y_r^2)
			      \le \sigma_1^2,
		      \]
		      pues \(\|y\|_2=1\). De aquí,
		      \[
			      \|Ax\|_2 \le \sigma_1
			      \quad \text{para todo } \|x\|_2=1,
		      \]
		      y por tanto \(\|A\|_2 \le \sigma_1\).

		      Para la desigualdad opuesta, tomamos \(x=Ve_1\), donde \(e_1=(1,0,\dots,0)\).
		      Entonces \(\|x\|_2=1\) y
		      \[
			      \|Ax\|_2=\|\Sigma e_1\|_2=\sigma_1.
		      \]
		      Concluimos que \(\|A\|_2=\sigma_1\).






		\item Dado que $\norm{A}_F = \sqrt{\tr(A^{\top}A)}$, se tiene que \[
			      A^{\top} = (U \Sigma V^{\top})^{\top} = V \Sigma^{\top} U^{\top},
		      \]
		      entonces, como $U$ es ortogonal:
		      \[
			      A^{\top} A
			      = (V \Sigma^{\top} U^{\top})(U \Sigma V^{\top})
			      = V \Sigma^{\top} \underbrace{U^{\top} U}_{I} \Sigma V^{\top}
			      = V \Sigma^{\top} \Sigma V^{\top}.
		      \]
		      Como la traza es invariante bajo cambios de base ortogonales, y la multiplicación de matrices es asociativa, podemos considerar que $V$ es un cambio de base ortogonal respecto a $\Sigma^{\top} \Sigma$. Así que:
		      \[ \tr(V \Sigma^{\top} \Sigma V^{\top}) = \tr(\Sigma^{\top} \Sigma) = \sum_{i=1}^{n} |\sigma_i|^2 = \sum_{i = 1}^{r} |\sigma_i|^2. \]
		      Luego $\norm{A}_F = \sqrt{\tr(\Sigma^{\top}\Sigma)} = \sqrt{\sigma_1^2 + \ldots + \sigma_r^2}$ (considerando que si $i > r$, $\sigma_i = 0$).
	\end{enumerate}
\end{proof}

\textit{Notación:} Llamaremos a la columna $i$-ésima de $U$ y $V$, $u_i$ y $v_i$ respectivamente.

\begin{theorem}[Eckart-Young]\label{thm:eckart-young}
	Sea $A \in M_{m \times n}(\R)$ una matriz de rango $r$, $A = U \Sigma V^{\top}$ su descomposición en valores singulares. Sea \[ A_k = \sum_{i=1}^{k}\sigma_i u_i v_i^{\top}. \]
	Entonces, para toda matriz $B \in M_{m \times n}(\R)$ de rango $k$, se tiene que:
	\[ \norm{A-A_k}_F \leq \norm{A-B}_F, \]
	\[ \norm{A-A_k}_2 \leq \norm{A-B}_2. \]
	Es decir, el teorema nos garantiza que $A_k$ es la mejor aproximación de rango $k$ a la matriz $A$ tanto en norma espectral como en norma de Frobenius.
\end{theorem}

\begin{proof}[Prueba (norma 2)]
	Primero probamos que $\norm{A - A_k}_2 = \sigma_{k+1}$. Para simplificar la notación, $\norm{\cdot}_2 = \norm{\cdot}$.\\

	A partir de $A = \sum_{i=1}^{r} \sigma_i u_i v_i^{\top}$ se obtiene
	\[
		A - A_k = \sum_{i=k+1}^{r} \sigma_i u_i v_i^{\top},
	\]
	que es una descomposición SVD válida de $A - A_k$, cuyos valores singulares son $\sigma_{k+1} \geq \cdots \geq \sigma_r > 0$. En particular, el mayor de ellos es $\sigma_{k+1}$, y por \cref{lem:normas-svd}:
	\[
		\norm{A - A_k}_2 = \sigma_{k+1}.
	\]

	Ahora procederemos por reducción al absurdo. Supongamos que existe $B \in M_{m \times n}(\R)$ con $\rk(B) = k$ tal que
	\[
		\norm{A - B}_2 < \norm{A - A_k}_2 = \sigma_{k+1}.
	\]
	Para cualquier $w \in \R^n$, tenemos que
	\[
		\norm{(A - B)w} < \sigma_{k+1}\norm{w}.
	\]
	De ello se sigue que, para cualquier $w \in \ker(B)$,
	\[
		\norm{(A - B)w} = \norm{Aw} < \sigma_{k+1}\norm{w}. \tag{1}
	\]

	Ahora, para cualquier $w \in V_{k+1} = \spn\{v_1, v_2, \ldots, v_{k+1}\}$, vamos a ver que
	\[
		\norm{Aw} \geq \sigma_{k+1}\norm{w}.
	\]
	Escribimos $w \in V_{k+1}$, como combinación lineal de los elementos de la base:
	\[ Aw
		= A\parens*{\sum_{i=1}^{k+1} c_i v_i}
		= \sum_{i=1}^{k+1} c_i A v_i. \]
	Asimismo, por definición de SVD,
	\[ A v_i = \sigma_i u_i. \]
	Desarrollando la norma tenemos que:
	\[ \norm{Aw}^2
		= \ip*{\sum_{i=1}^{k+1} c_i\sigma_i u_i,
			\sum_{j=1}^{k+1} c_j\sigma_j u_j}
		= \sum_{i=1}^{k+1}\sum_{j=1}^{k+1} c_i \sigma_i c_j \sigma_j \ip{u_i, u_j}. \]
	Como los $u_i$ son ortonormales, entonces $\ip{u_i, u_j} = \delta_{ij}$, así que desaparecen los términos cruzados:
	\[
		= \sum_{i=1}^{k+1}\sum_{j=1}^{k+1} c_i \sigma_i c_j \sigma_j \delta_{ij}
		= \sum_{i=1}^{k+1} c_i^2 \sigma_i^2.
	\]
	Como $\sigma_i \geq \sigma_{k+1}$ para $i \leq k+1$,
	\[ \sum_{i=1}^{k+1} c_i^2 \sigma_i^2
		\geq
		\sigma_{k+1}^2 \sum_{i=1}^{k+1} c_i^2
		=
		\sigma_{k+1}^2 \norm{w}^2. \]
	Así que
	\[
		\norm{Aw} \geq \sigma_{k+1}\norm{w}. \tag{2}
	\]

	Por el teorema rango-nulidad aplicado a $B$, que tiene rango $k$,
	\[
		\dim(\ker(B)) = n - k.
	\]
	Por otro lado, $V_{k+1} = \spn\{v_1, \ldots, v_{k+1}\}$, así que
	\[
		\dim(V_{k+1}) = k + 1.
	\]
	Ambos son subespacios de $\R^n$, por lo que su suma también lo es:
	\[
		\ker(B) + V_{k+1} \subseteq \R^n
		\implies
		\dim(\ker(B) + V_{k+1}) \leq n.
	\]
	Aplicando la fórmula de Grassmann a estos dos subespacios,
	\[
		\dim(\ker(B) + V_{k+1}) = \dim(\ker(B)) + \dim(V_{k+1}) - \dim(\ker(B) \cap V_{k+1}),
	\]
	y despejando la dimensión de la intersección,
	\[
		\dim(\ker(B) \cap V_{k+1})
		= \dim(\ker(B)) + \dim(V_{k+1}) - \dim(\ker(B) + V_{k+1}).
	\]
	Como $\dim(\ker(B) + V_{k+1}) \leq n$ obtenemos
	\[
		\dim(\ker(B) \cap V_{k+1})
		\geq \dim(\ker(B)) + \dim(V_{k+1}) - n
		= (n-k) + (k+1) - n = 1.
	\]
	Como la intersección tiene dimensión al menos $1$, no es el subespacio $\{0\}$, y por tanto existe $w \neq 0$ tal que $w \in \ker(B) \cap V_{k+1}$.
	Sin embargo, las desigualdades $(1)$ y $(2)$ no pueden cumplirse simultáneamente, lo que lleva a una contradicción.
\end{proof}

Para la prueba en norma de Frobenius será necesario emplear el siguiente lema:

\begin{lemma}[Weyl]\label{lem:weyl}
	Sea $A \in M_{m \times n}(\R)$, descompuesta en una suma $A = B + C$. Denotamos por $\sigma_j(M)$ el valor singular $j$-ésimo de una matriz cualquiera, en este caso $M = A, B, C$. Sea además $\sigma_j(M) = 0$ para cualquier $j > \rk(M)$. Se tiene que
	\[ \sigma_{i+j-1}(A) \leq \sigma_i(B) + \sigma_j(C) \quad \forall i, j \geq 1. \]
\end{lemma}

\begin{proof}
	Sean $i, j \in \N$ arbitrarios, con $1 \leq i, j \leq n$. Recordamos que $\norm{X - X_k}_2 = \sigma_{k+1}$.
	\[ \sigma_i(B) + \sigma_j(C) = \norm{B - B_{i-1}}_2 + \norm{C - C_{j-1}}_2 = \sigma_1(B - B_{i-1}) + \sigma_1(C - C_{j-1}). \]
	Utilizando la desigualdad triangular, se tiene que:
	\[ \norm{B - B_{i-1}}_2 + \norm{C - C_{j-1}}_2 \geq \norm{B - B_{i-1} + C - C_{j-1}}_2 = \norm{A - B_{i-1} - C_{j-1}}_2. \]
	Por otra parte, recordemos que el rango de una matriz coincide con la dimensión de su imagen. Entonces, dado que
	\[
		\im(X + Y) \subseteq \im(X) + \im(Y),
	\]
	se tiene
	\begin{align*}
		\rk(X + Y) = \dim(\im(X+Y))
		 & \leq \dim(\im(X) + \im(Y))                          \\
		 & \leq \dim(\im(X)) + \dim(\im(Y)) = \rk(X) + \rk(Y),
	\end{align*}
	que se deduce de la fórmula de Grassmann. \\

	El resultado anterior nos indica que:
	\[ \rk\parens*{B_{i-1} + C_{j-1}} \leq i + j - 2. \]

	Por el teorema de Eckart-Young para la norma $2$:
	\[ \norm{A - B_{i-1} - C_{j-1}}_2 \geq \norm{A - A_{i+j-2}}_2 = \sigma_{i+j-1}(A). \]
\end{proof}

\begin{proof}[Prueba (norma de Frobenius)]
	Por \cref{lem:weyl}, para cualquier matriz $M$ de rango $k$, y para todo $i = 1, \ldots, r - k$, se tiene que:
	\[ \sigma_{k+i}(A) \leq \sigma_i(A - M) + \sigma_{k+1}(M) = \sigma_i(A - M), \tag{$\ast$} \]
	ya que $\sigma_{k+1}(M) = 0$ puesto que el rango de $M$ es $k$. Recordamos por \cref{lem:normas-svd} que $\norm{A}_F = \sqrt{\sum_{i=1}^{r} \sigma_i^2}$.
	Tenemos que:
	\[ \norm{A - M}_F^2 = \sum_{i=1}^{\min(m,n)} \sigma_i^2(A - M) \geq \sum_{i=1}^{r-k} \sigma_i^2(A - M). \]
	Aplicando $(\ast)$,
	\[ \norm{A - M}_F^2 \geq \sum_{i=1}^{r-k} \sigma_i^2(A - M) \geq \sum_{i=1}^{r-k} \sigma_{k+i}^2(A) = \sum_{j=k+1}^{r} \sigma_j^2(A) = \norm{A - A_k}_F^2. \]
\end{proof}

\section{Aplicación práctica: compresión y representación de imágenes mediante SVD}
Antes de entrar en detalles técnicos, conviene saber que una imagen digital admite una interpretación matemática sorprendentemente sencilla.\\
Para simplificar la exposición, nos centraremos en imágenes en escala de grises. A diferencia de las imágenes en color, que requieren tres matrices separadas para representar los canales rojo, verde y azul (RGB), una imagen en escala de grises queda completamente descrita por una única matriz, lo que permite ilustrar las ideas fundamentales de este teorema.\\
Si imaginamos dicha imagen como una cuadrícula de pequeños puntos (los píxeles de la pantalla), cada uno de ellos lleva asociado un único valor numérico que indica su nivel de gris, desde el negro absoluto hasta el blanco puro. Al organizar estos valores según la posición de cada píxel en la imagen, fila a fila y columna a columna, obtenemos una tabla de números: una matriz.\\
De este modo, cada entrada de la matriz se corresponde exactamente con un píxel de la imagen.\\
El problema surge cuando la imagen tiene alta resolución: su matriz asociada puede contener millones de entradas, lo que exige una cantidad considerable de memoria para almacenarla o transmitirla. Aquí es donde el teorema de Eckart-Young resulta de gran utilidad, pues nos garantiza que es posible obtener una aproximación de calidad aceptable truncando la descomposición en valores singulares (SVD) de dicha matriz.\\
El siguiente programa de MATLAB aplica precisamente este principio: reconstruye una versión aproximada de la imagen original con un rango reducido.
\begin{verbatim}
X = imread('quijote.jpeg');   % Leer imagen

X = rgb2gray(X);              % Convertir a escala de grises (matriz 2D)

A = im2double(X);             % Normalizar valores a [0,1]

[U,S,V] = svd(A);             % Descomposición SVD

k = 25;                       % Número de valores singulares

Ak = U(:,1:k) * S(1:k,1:k) * V(:,1:k)';

imshow(Ak)
colormap(gray)
\end{verbatim}

Se observa que no es necesario utilizar un número elevado de valores singulares para obtener una reconstrucción de calidad aceptable.
Teniendo en cuenta que la imagen original tiene dimensiones $891 \times 1260$, lo que implica un máximo de $891$ valores singulares, se obtiene una aproximación visualmente muy fiel utilizando únicamente $k=100$.
Esto supone una reducción significativa en la cantidad de información necesaria para representar la imagen.

Cabe preguntarse qué papel desempeñan los vectores de las bases $U$ y $V$ en la imagen reconstruida. Cada vector $u_i \in \mathbb{R}^m$ es un vector singular izquierdo: sus $m$ componentes indexan las filas de $A$ y describen un patrón de variación vertical. De manera análoga, cada vector $v_i \in \mathbb{R}^n$ es un vector singular derecho: sus $n$ componentes indexan las columnas y describen un patrón de variación horizontal.

El producto $\sigma_i u_i v_i^{\top}$ constituye una matriz de rango 1, que llamaremos \emph{modo},
cuyo píxel en la coordenada $(p,q)$ está dado por $\sigma_i\, u_i(p)\, v_i(q)$, donde $u_i(p)$ y $v_i(q)$ representan las componentes $p$-ésima y $q$-ésima de los vectores $u_i$ y $v_i$, respectivamente.
Con el propósito de ilustrar este fenómeno, a continuación se analiza una imagen simplificada para examinar, de forma individual,
el aporte visual de cada una de estas componentes. En este
caso consideraremos que 0 corresponde al blanco puro y 1 al negro puro.

Es decir, la reconstrucción de la imagen se obtiene mediante una suma de contribuciones de rango 1.
En particular, para cada posición (p,q), el valor final del píxel se calcula como la suma de las aportaciones de todos los modos en dicha posición.
